TiDB Development Guide

Optimistic Transaction

Under the optimistic transaction model, modification conflicts are regarded as part of the transaction commit. TiDB cluster uses the optimistic transaction model by default before version 3.0.8, uses pessimistic transaction model after version 3.0.8. System variable tidb_txn_mode controls TiDB uses optimistic transaction mode or pessimistic transaction mode.

This document talks about the implementation of optimistic transaction in TiDB. It is recommended that you have learned about the principles of optimistic transaction.

This document refers to the code of TiDB v5.2.1

Check the syntax "start transaction read only" and "as of timestamp" used correctly.
If stale read timestamp was set,  creates a new stale read transaction and sets "in transaction" state, and return.
create a new transaction and set some properties like snapshot, startTS etc
*/

func (e *SimpleExec) executeBegin(ctx context.Context, s *ast.BeginStmt) error {
‚Äã    if s.ReadOnly {
‚Äã       // the statement "start transaction read only" must be used with tidb_enable_noop_functions is true
‚Äã       //  the statement "start transaction read only  as of timestamp" can be used Whether  tidb_enable_noop_functions  is true or falseÔºåbut that tx_read_ts mustn't be set.
‚Äã       //  the statement "start transaction read only  as of timestamp" must ensure the timestamp is in the legal safe point range        
‚Äã       enableNoopFuncs := e.ctx.GetSessionVars().EnableNoopFuncs
‚Äã       if !enableNoopFuncs && s.AsOf == nil {
‚Äã           return expression.ErrFunctionsNoopImpl.GenWithStackByArgs("READ ONLY")
‚Äã       }

‚Äã       if s.AsOf != nil {
‚Äã           if e.ctx.GetSessionVars().TxnReadTS.PeakTxnReadTS() > 0 {
‚Äã              return errors.New("start transaction read only as of is forbidden after set transaction read only as of")
‚Äã           }
‚Äã       }
‚Äã    } 

‚Äã    // process stale read transaction
‚Äã    if e.staleTxnStartTS > 0 {
‚Äã      // check timestamp of stale read correctly
‚Äã       if err := e.ctx.NewStaleTxnWithStartTS(ctx, e.staleTxnStartTS); err != nil {
‚Äã           return err
‚Äã       }

‚Äã       // ignore tidb_snapshot configuration if in stale read transaction
‚Äã       vars := e.ctx.GetSessionVars()
‚Äã       vars.SetSystemVar(variable.TiDBSnapshot, "")
    
‚Äã       // set "in transaction" state and return
‚Äã       vars.SetInTxn(true)

‚Äã       return nil
‚Äã    }

‚Äã    /* If BEGIN is the first statement in TxnCtx, we can reuse the existing transaction, without the need to call NewTxn, which commits the existing transaction and begins a new one. If the last un-committed/un-rollback transaction is a time-bounded read-only transaction, we should always create a new transaction. */
    
‚Äã    txnCtx := e.ctx.GetSessionVars().TxnCtx
‚Äã    if txnCtx.History != nil || txnCtx.IsStaleness {
‚Äã       err := e.ctx.NewTxn(ctx)
‚Äã    }

‚Äã  // set "in transaction" state
   e.ctx.GetSessionVars().SetInTxn(true)

   // create a new transaction and set some properties like snapshot, startTS etc.
   txn, err := e.ctx.Txn(true)
‚Äã    // set Linearizability option
‚Äã    if s.CausalConsistencyOnly {
‚Äã       txn.SetOption(kv.GuaranteeLinearizability, false)
‚Äã    }

‚Äã    return nil
}

DML Executed In Optimistic Transaction

There are three kinds of DML operations, such as update, delete and insert. For simplicity, This article only describes the update statement execution process. DML mutations are not sended to tikv directly, but buffered in TiDB temporarily, commit operation make the mutations effective at last.

prewrite

The entry function to prewrite a transaction is (c *twoPhaseCommitter) prewriteMutations which calls the function (batchExe *batchExecutor) process to do it. The function (batchExe *batchExecutor) process calls (batchExe *batchExecutor) startWorker to prewrite evenry batch parallelly. The function (batchExe *batchExecutor) startWorker calls (action actionPrewrite) handleSingleBatch to prewrite a single batch.

(batchExe *batchExecutor) process

The important comment and simplified code are as followers. The completed code is here .

// start worker routine to prewrite every batch parallely and collect results
func (batchExe *batchExecutor) process(batches []batchMutations) error {
‚Äã    var err error
‚Äã    err = batchExe.initUtils()
‚Äã    // For prewrite, stop sending other requests after receiving first error.
‚Äã    var cancel context.CancelFunc

‚Äã    if _, ok := batchExe.action.(actionPrewrite); ok {
‚Äã       batchExe.backoffer, cancel = batchExe.backoffer.Fork()
‚Äã       defer cancel()  
‚Äã    }

‚Äã    // concurrently do the work for each batch.
‚Äã    ch := make(chan error, len(batches))
‚Äã    exitCh := make(chan struct{})
‚Äã    go batchExe.startWorker(exitCh, ch, batches)
‚Äã    // check results of every batch prewrite synchronously, if one batch fails, 
‚Äã    // stops every prewrite worker routine immediately.
‚Äã    for i := 0; i < len(batches); i++ {
‚Äã       if e := <-ch; e != nil {
‚Äã           // Cancel other requests and return the first error.
‚Äã           if cancel != nil {
‚Äã              cancel()
‚Äã           }

‚Äã           if err == nil {
‚Äã              err = e
‚Äã           }
‚Äã       }
‚Äã    }

‚Äã    close(exitCh)   // break the loop of function startWorker

‚Äã    return err
}

(action actionPrewrite) handleSingleBatch

The important comment and simplified code are as followers. The completed code is here .

/*

create a prewrite request and a region request sender that sends the prewrite request to tikv.
(1)get Prewrite Response coming from tikv
(2)If no error happened and it is OnePC transaction, update onePCCommitTS by prewriteResp and return
(3)if no error happened and it is AsyncCommit transaction, update minCommitTS  if need and return
(4)If errors hanpped beacause of lock confilict, extract the locks from the error responsed, resolove the locks expired
(5)do the backoff for prewrite
*/

func (action actionPrewrite) handleSingleBatch(c *twoPhaseCommitter, bo *retry.Backoffer, batch batchMutations) (err error) {
‚Äã    // create a prewrite request and a region request sender that sends the prewrite request to tikv.
‚Äã    txnSize := uint64(c.regionTxnSize[batch.region.GetID()])
‚Äã    req := c.buildPrewriteRequest(batch, txnSize)
‚Äã    sender := locate.NewRegionRequestSender(c.store.GetRegionCache(), c.store.GetTiKVClient())

‚Äã    for {
‚Äã       resp, err := sender.SendReq(bo, req, batch.region, client.ReadTimeoutShort)
‚Äã       regionErr, err := resp.GetRegionError()  

‚Äã       // get Prewrite Response coming from tikv
‚Äã       prewriteResp := resp.Resp.(*kvrpcpb.PrewriteResponse)
‚Äã       keyErrs := prewriteResp.GetErrors()
‚Äã       if len(keyErrs) == 0 {  
‚Äã           // If no error happened and it is OnePC transaction, update onePCCommitTS by prewriteResp and return
‚Äã           if c.isOnePC() {
‚Äã                c.onePCCommitTS = prewriteResp.OnePcCommitTs
‚Äã                return nil
‚Äã          } 

‚Äã          // if no error happened and it is AsyncCommit transaction, update minCommitTS  if need and return
‚Äã          if c.isAsyncCommit() {
‚Äã                  if prewriteResp.MinCommitTs > c.minCommitTS {
‚Äã                       c.minCommitTS = prewriteResp.MinCommitTs
‚Äã                  }
‚Äã           }
‚Äã           return nil
‚Äã       }// if len(keyErrs) == 0

‚Äã       // If errors hanpped beacause of lock confilict, extract the locks from the error responsed
‚Äã       var locks []*txnlock.Lock
‚Äã       for _, keyErr := range keyErrs {
‚Äã           // Extract lock from key error
‚Äã           lock, err1 := txnlock.ExtractLockFromKeyErr(keyErr)
‚Äã           if err1 != nil {
‚Äã              return errors.Trace(err1)
‚Äã           }
‚Äã           locks = append(locks, lock)
‚Äã       }// for _, keyErr := range keyErrs
‚Äã       // resolve conflict locks expired, do the backoff for prewrite
‚Äã       start := time.Now()
‚Äã       msBeforeExpired, err := c.store.GetLockResolver().ResolveLocksForWrite(bo, c.startTS, c.forUpdateTS, locks)
‚Äã       if msBeforeExpired > 0 {
‚Äã           err = bo.BackoffWithCfgAndMaxSleep(retry.BoTxnLock, int(msBeforeExpired), errors.Errorf("2PC prewrite lockedKeys: %d", len(locks)))
‚Äã           if err != nil {
‚Äã              return errors.Trace(err) // backoff exceeded maxtime, returns error
‚Äã           }
‚Äã       }
‚Äã    }// for loop
}

commit

The entry function of commiting a transaction is (c *twoPhaseCommitter) commitMutations which calls the function (c *twoPhaseCommitter) doActionOnGroupMutations to do it. The batch of primary key will be committed first, then the function (batchExe *batchExecutor) process calls (batchExe *batchExecutor) startWorker to commit the rest batches parallelly and asynchronously. The function (batchExe *batchExecutor) startWorker calls (actionCommit) handleSingleBatch to commit a single batch.

(c *twoPhaseCommitter) doActionOnGroupMutations

The important comment and simplified code are as followers. The completed code is here .

/*
If the groups contain primary, commit the primary batch synchronously
If the first time to commit, spawn a goroutine to commit secondary batches asynchronously
if retry to commit,  commit the secondary batches synchronously, because itself is in the asynchronously goroutine
*/

func (c *twoPhaseCommitter) doActionOnGroupMutations(bo *retry.Backoffer, action twoPhaseCommitAction, groups []groupedMutations) error {
‚Äã    batchBuilder := newBatched(c.primary())
‚Äã    // Whether the groups being operated contain primary
‚Äã    firstIsPrimary := batchBuilder.setPrimary()
‚Äã    actionCommit, actionIsCommit := action.(actionCommit)
‚Äã    c.checkOnePCFallBack(action, len(batchBuilder.allBatches()))
‚Äã    // If the groups contain primary, commit the primary batch synchronously
‚Äã    if firstIsPrimary &&
‚Äã       (actionIsCommit && !c.isAsyncCommit()) {
‚Äã         // primary should be committed(not async commit)/cleanup/pessimistically locked first
‚Äã       err = c.doActionOnBatches(bo, action, batchBuilder.primaryBatch())    
‚Äã       batchBuilder.forgetPrimary()
‚Äã    }
‚Äã  // If the first time to commit, spawn a goroutine to commit secondary batches asynchronously
   // if retry to commit,  commit the secondary batches synchronously, because itself is in the asynchronously goroutine
‚Äã    if actionIsCommit && !actionCommit.retry && !c.isAsyncCommit() {
‚Äã        secondaryBo := retry.NewBackofferWithVars(c.store.Ctx(), CommitSecondaryMaxBackoff, c.txn.vars)
‚Äã       c.store.WaitGroup().Add(1)
‚Äã       go func() {
‚Äã           defer c.store.WaitGroup().Done()
‚Äã           e := c.doActionOnBatches(secondaryBo, action, batchBuilder.allBatches())
‚Äã       }
‚Äã    }
‚Äã    else {
‚Äã       err = c.doActionOnBatches(bo, action, batchBuilder.allBatches())
‚Äã    }

‚Äã    return errors.Trace(err)
}

(actionCommit) handleSingleBatch

The function (batchExe *batchExecutor) process calls the function (actionCommit) handleSingleBatch to send commit request to all tikv nodes.

/*
create a commit request and commit sender.
If regionErr happened, backoff and retry the commit operation.
If the error is not a regionErr, but rejected by TiKV beacause the commit ts was expired, retry with a newer commits.
Other errors happened, return error immediately.
No error happened, exit the for loop and return success.
*/

func (actionCommit) handleSingleBatch(c *twoPhaseCommitter, bo *retry.Backoffer, batch batchMutations) error {
‚Äã    // create a commit request and commit sender
‚Äã    keys := batch.mutations.GetKeys()
‚Äã    req := tikvrpc.NewRequest(tikvrpc.CmdCommit, &kvrpcpb.CommitRequest{
‚Äã       StartVersion: c.startTS,
‚Äã        Keys:     keys,
‚Äã       CommitVersion: c.commitTS,
‚Äã    }, kvrpcpb.Context{Priority: c.priority, SyncLog: c.syncLog,
‚Äã       ResourceGroupTag: c.resourceGroupTag, DiskFullOpt: c.diskFullOpt}) 

‚Äã    tBegin := time.Now()
‚Äã    attempts := 0 
‚Äã    sender := locate.NewRegionRequestSender(c.store.GetRegionCache(), c.store.GetTiKVClient())

‚Äã    for {
‚Äã       attempts++
‚Äã       resp, err := sender.SendReq(bo, req, batch.region, client.ReadTimeoutShort)
‚Äã       regionErr, err := resp.GetRegionError()
‚Äã       // If regionErr happened, backoff and retry the commit operation
‚Äã       if regionErr != nil {
‚Äã           // For other region error and the fake region error, backoff because there's something wrong.
‚Äã           // For the real EpochNotMatch error, don't backoff.
‚Äã           if regionErr.GetEpochNotMatch() == nil || locate.IsFakeRegionError(regionErr) {
‚Äã              err = bo.Backoff(retry.BoRegionMiss, errors.New(regionErr.String()))
‚Äã              if err != nil {
‚Äã                  return errors.Trace(err)
‚Äã              }
‚Äã           }

‚Äã           same, err := batch.relocate(bo, c.store.GetRegionCache())
‚Äã           if err != nil {
‚Äã              return errors.Trace(err)
‚Äã           }

‚Äã           if same {
‚Äã              continue
‚Äã           }

‚Äã           err = c.doActionOnMutations(bo, actionCommit{true}, batch.mutations)
‚Äã           return errors.Trace(err)
‚Äã       }// if regionErr != nil

‚Äã	   // If the error is not a regionErr, but rejected by TiKV beacause the commit ts was expired, retry with a newer commits. Other errors happened, return error immediately.

‚Äã       commitResp := resp.Resp.(*kvrpcpb.CommitResponse)
‚Äã       if keyErr := commitResp.GetError(); keyErr != nil {
‚Äã           if rejected := keyErr.GetCommitTsExpired(); rejected != nil {
‚Äã              // 2PC commitTS rejected by TiKV, retry with a newer commits, update commit ts and retry.
‚Äã              commitTS, err := c.store.GetTimestampWithRetry(bo, c.txn.GetScope())
‚Äã              c.mu.Lock()
‚Äã              c.commitTS = commitTS
‚Äã              c.mu.Unlock()
‚Äã              // Update the commitTS of the request and retry.
‚Äã              req.Commit().CommitVersion = commitTS
‚Äã              continue
‚Äã           }

‚Äã           if c.mu.committed {
‚Äã              // 2PC failed commit key after primary key committed
‚Äã              // No secondary key could be rolled back after it's primary key is committed.
‚Äã               return errors.Trace(err)
‚Äã           }
‚Äã           return err
‚Äã       }

‚Äã        // No error happened, exit the for loop
‚Äã        break  
‚Äã    }// for loop
}
